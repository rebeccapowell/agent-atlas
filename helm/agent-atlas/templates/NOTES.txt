Agent Atlas has been deployed!

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}/
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "agent-atlas.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "agent-atlas.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward service/{{ include "agent-atlas.fullname" . }} 8080:{{ .Values.service.port }}
  Visit http://localhost:8080
{{- end }}

2. Check health:
  kubectl --namespace {{ .Release.Namespace }} exec deploy/{{ include "agent-atlas.fullname" . }} -- curl -s http://localhost:8080/healthz

3. View tools catalog:
  kubectl --namespace {{ .Release.Namespace }} port-forward service/{{ include "agent-atlas.fullname" . }} 8080:{{ .Values.service.port }}
  curl http://localhost:8080/v1/tools

4. Configure OIDC:
  helm upgrade {{ .Release.Name }} ./helm/agent-atlas \
    --set atlas.oidc.issuer=https://your-idp.example.com \
    --set atlas.oidc.audience=api://agent-atlas
